 46     /**
 47      * Sets up some standard variables as well as starts the user agent parsing process
 48      *
 49      * @return {Object}       the result of the user agent parsing
 50      */
 51     public static function parse($ua = NULL) {
 52 
 53         self::$ua      = $ua ? $ua : strip_tags($_SERVER["HTTP_USER_AGENT"]);
 54         self::$accept  = empty($_SERVER["HTTP_ACCEPT"]) ? '' : strip_tags($_SERVER["HTTP_ACCEPT"]);
 55         if (empty(self::$regexes)) {
 56             if (file_exists(__DIR__."/resources/regexes.yaml")) {
 57                 self::$regexes = yaml_parse_file(__DIR__."/resources/regexes.yaml");
 58             } else {
 59                 print "<h1>Error</h1>
 60                     <p>Please download the regexes.yaml file before using UAParser.php.</p>
 61                     <p>You can type the following at the command line to download the latest version:</p>
 62                     <blockquote>
 63                     <code>%: cd /path/to/UAParser</code><br />
 64                     <code>%: php UAParser.php -get</code>
 65                     </blockquote>";
 66                 exit;
 67             }
 68         }
 69 
 70 
 71         // run the regexes to match things up
 72         $uaRegexes = self::$regexes['user_agent_parsers'];
 73         foreach ($uaRegexes as $uaRegex) {
 74             if ($result = self::uaParser($uaRegex)) {
 75                 $result->uaOriginal = self::$ua;
 76                 break;
 77             }
 78         }
 79 
 80         // if no browser was found check to see if it can be matched at least against a device (e.g. spider, generic feature phone or generic smartphone)
 81         if (!$result) {
 82             if (($result = self::deviceParser()) && ($result->device != 'Spider')) {
 83                 $result->isMobile       = true;
 84                 $result->isMobileDevice = true; 
 85                 $result->uaOriginal     = self::$ua;
 86             } else if (isset($result) && isset($result->device) && ($result->device == "Spider")) {
 87                 $result->isMobile       = false;
 88                 $result->isSpider       = true;
 89                 $result->uaOriginal     = self::$ua;
 90             }
 91         }
 92 
 93         // still false?! see if it's a really dumb feature phone, if not just mark it as unknown
 94         if (!$result) {
 95             if ((strpos(self::$accept,'text/vnd.wap.wml') > 0) || (strpos(self::$accept,'application/vnd.wap.xhtml+xml') > 0) || isset($_SERVER['HTTP_X_WAP_PROFILE']) || isset($_SERVER['HTTP_PROFILE'])) {
 96                 $result = new \stdClass();
 97                 $result->device         = "Generic Feature Phone";
 98                 $result->deviceFull     = "Generic Feature Phone";
 99                 $result->isMobile       = true;
100                 $result->isMobileDevice = true;
101                 $result->uaOriginal     = self::$ua;
102             } else {
103                 $result = new \stdClass();
104                 $result->device         = "Unknown";
105                 $result->deviceFull     = "Unknown";
106                 $result->isMobile       = false;
107                 $result->isMobileDevice = false;
108                 $result->isComputer     = true;
109                 $result->uaOriginal     = self::$ua;
110             }
111         }
112 
113         // log the results when testing
114         if (self::$debug) {
115             self::log($result);
116         }
117 
118         return $result;
119     }

122     /**
123      * Attemps to see if the user agent matches the regex for this test. If so it populates an obj
124      * with properties based on the user agent. Will also try to fetch OS & device properties
125      *
126      * @param  {Array}        the regex to be tested as well as any extra variables that need to be swapped
127      *
128      * @return {Object}       the result of the user agent parsing
129      */
130     private static function uaParser($regex) {
131 
132         // tests the supplied regex against the user agent
133         if (preg_match("/".str_replace("/","\/",$regex['regex'])."/",self::$ua,$matches)) {
134 
135             // Define safe parser defaults
136             $defaults = array(
137                 'isMobileDevice' => false,
138                 'isMobile' => false,
139                 'isSpider' => false,
140                 'isTablet' => false,
141                 'isComputer' => true,
142             );
143             // build the obj that will be returned starting with defaults
144             $obj = (object) $defaults;
145 
146             // build the version numbers for the browser
147             if (isset($matches[2]) || isset($regex['v1_replacement'])) {
148                 $obj->major  = isset($regex['v1_replacement']) ? $regex['v1_replacement'] : $matches[2];
149             } else {
150                 $obj->major = '';
151             }
152             if (isset($matches[3]) || isset($regex['v2_replacement'])) {
153                 $obj->minor = isset($regex['v2_replacement']) ? $regex['v2_replacement'] : $matches[3];
154             }
155             if (isset($matches[4])) {
156                 $obj->build = $matches[4];
157                 $obj->patch = $matches[4];
158             }
159             if (isset($matches[5])) {
160                 $obj->revision = $matches[5];
161             }
162 
163             // pull out the browser family. replace the version number if necessary
164             $obj->browser = isset($regex['family_replacement']) ? str_replace("$1",$obj->major,$regex['family_replacement']) : $matches[1];
165             $obj->family  = isset($regex['family_replacement']) ? str_replace("$1",$obj->major,$regex['family_replacement']) : $matches[1];
166 
167             // set-up a clean version number
168             $obj->version = isset($obj->major) ? $obj->major : "";
169             $obj->version = isset($obj->minor) ? $obj->version.'.'.$obj->minor : $obj->version;
170             $obj->version = isset($obj->build) ? $obj->version.'.'.$obj->build : $obj->version;
171             $obj->version = isset($obj->revision) ? $obj->version.'.'.$obj->revision : $obj->version;
172 
173             // prettify
174             $obj->browserFull = $obj->browser;
175             if ($obj->version != '') {
176                 $obj->browserFull .= " ".$obj->version;
177             }
178 
179             // detect if this is a uiwebview call on iOS
180             $obj->isUIWebview = (($obj->browser == 'Mobile Safari') && !strstr(self::$ua,'Safari')) ? true : false;
181 
182             // check to see if this is a mobile browser
183             $mobileBrowsers = array("Firefox Mobile","Opera Mobile","Opera Mini","Mobile Safari","webOS","IE Mobile","Playstation Portable",
184                 "Nokia","Blackberry","Palm","Silk","Android","Maemo","Obigo","Netfront","AvantGo","Teleca","SEMC-Browser",
185                 "Bolt","Iris","UP.Browser","Symphony","Minimo","Bunjaloo","Jasmine","Dolfin","Polaris","BREW","Chrome Mobile",
186                 "UC Browser","Tizen Browser");
187             foreach($mobileBrowsers as $mobileBrowser) {
188                 if (stristr($obj->browser, $mobileBrowser)) {
189                     $obj->isMobile = true;
190                     break;
191                 }
192             }
193 
194             // figure out the OS for the browser, if possible
195             if ($osObj = self::osParser()) {
196                 $obj = (object) array_merge((array) $obj, (array) $osObj);
197             }
198 
199             // create an attribute combinining browser and os
200             if (isset($obj->osFull) && $obj->osFull) {
201                 $obj->full = $obj->browserFull."/".$obj->osFull;
202             }
203 
204             // figure out the device name for the browser, if possible
205             if ($deviceObj = self::deviceParser()) {
206                 $obj = (object) array_merge((array) $obj, (array) $deviceObj);
207             }
208 
209             // if this is a mobile browser make sure mobile device is set to true
210             if ($obj->isMobile) {
211                 $obj->isMobileDevice = true; // this is going to catch android devices
212             } else if ($obj->isMobileDevice) {
213                 $obj->isMobile = true;       // this will catch some weird htc devices
214             }
215 
216             // if OS is Android check to see if this is a tablet. won't work on UA strings less than Android 3.0
217             // based on: http://googlewebmastercentral.blogspot.com/2011/03/mo-better-to-also-detect-mobile-user.html
218             // opera doesn't follow this convention though...
219             if ((isset($obj->os) && $obj->os == 'Android') && !strstr(self::$ua, 'Mobile') && !strstr(self::$ua, 'Opera')) {
220                 $obj->isTablet = true;
221                 $obj->isMobile = false;
222             }
223 
224             // some select mobile OSs report a desktop browser. make sure we note they're mobile
225             $mobileOSs = array('Windows Phone 6.5','Windows CE','Symbian OS');
226             if (isset($obj->os) && in_array($obj->os,$mobileOSs)) {
227                 $obj->isMobile       = true;
228                 $obj->isMobileDevice = true;
229             }
230 
231             if (stristr(self::$ua,"tablet")) {
232                 $obj->isTablet       = true;
233                 $obj->isMobileDevice = true;
234                 $obj->isMobile       = false;
235             }
236 
237             // record if this is a spider
238             $obj->isSpider = (isset($obj->device) && $obj->device == "Spider") ? true : false;
239 
240             // record if this is a computer
241             $obj->isComputer = (!$obj->isMobile && !$obj->isSpider && !$obj->isMobileDevice) ? true : false;
242 
243             return $obj;
244 
245         } else {
246             return false;
247         }
248     }

250     /**
251      * If the user agent is matched in uaParser() it also tries to check the OS and get properties
252      *
253      * @return {Object}       the result of the os parsing
254      */
255     private static function osParser() {
256 
257         // build the obj that will be returned
258         $osObj = new \stdClass();
259 
260         // run the regexes to match things up
261         $osRegexes = self::$regexes['os_parsers'];
262         foreach ($osRegexes as $osRegex) {
263             if (preg_match("/".str_replace("/","\/",$osRegex['regex'])."/",self::$ua,$matches)) {
264 
265                 // Make sure matches 2 and 3 are at least set to null for setting
266                 // Major and Minor defaults
267                 if (!isset($matches[1])) { $matches[1] = null; }
268                     if (!isset($matches[2])) { $matches[2] = null; }
269                         if (!isset($matches[3])) { $matches[3] = null; }
270 
271                             // basic properties
272                             $osObj->osMajor   = isset($osRegex['os_v1_replacement']) ? $osRegex['os_v1_replacement'] : $matches[2];
273                 $osObj->osMinor   = isset($osRegex['os_v2_replacement']) ? $osRegex['os_v2_replacement'] : $matches[3];
274                 if (isset($matches[4])) {
275                     $osObj->osBuild = $matches[4];
276                     $osObj->osPatch = $matches[4];
277                 }
278                 if (isset($matches[5])) {
279                     $osObj->osRevision = $matches[5];
280                 }
281                 $osObj->os        = isset($osRegex['os_replacement'])    ? str_replace("$1",$osObj->osMajor,$osRegex['os_replacement'])  : $matches[1];
282 
283                 // os version
284                 $osObj->osVersion = isset($osObj->osMajor) ? $osObj->osMajor : "";
285                 $osObj->osVersion = isset($osObj->osMinor) ? $osObj->osVersion.'.'.$osObj->osMinor : $osObj->osVersion;
286                 $osObj->osVersion = isset($osObj->osBuild) ? $osObj->osVersion.'.'.$osObj->osBuild : $osObj->osVersion;
287                 $osObj->osVersion = isset($osObj->osRevision) ? $osObj->osVersion.'.'.$osObj->osRevision : $osObj->osVersion; 
288 
289                 // prettify
290                 $osObj->osFull    = $osObj->os." ".$osObj->osVersion;
291 
292                 return $osObj;
293             }
294         }
295         return false;
296     }

298     /**
299      * If the user agent is matched in uaParser() it also tries to check the device and get its properties
300      *
301      * @return {Object}       the result of the device parsing
302      */
303     private static function deviceParser() {
304 
305         // build the obj that will be returned
306         $deviceObj = new \stdClass();
307 
308         // run the regexes to match things up
309         $deviceRegexes = self::$regexes['device_parsers'];
310         foreach ($deviceRegexes as $deviceRegex) {
311             if (preg_match("/".str_replace("/","\/",$deviceRegex['regex'])."/i",self::$ua,$matches)) {
312 
313                 // Make sure device matches are null
314                 // Device Name, Major and Minor defaults
315                 if (!isset($matches[1])) { $matches[1] = null; }
316                     if (!isset($matches[2])) { $matches[2] = null; }
317                         if (!isset($matches[3])) { $matches[3] = null; }
318 
319                             // basic properties
320                             $deviceObj->deviceMajor  = isset($deviceRegex['device_v1_replacement']) ? $deviceRegex['device_v1_replacement'] : $matches[2];
321                 $deviceObj->deviceMinor  = isset($deviceRegex['device_v2_replacement']) ? $deviceRegex['device_v2_replacement'] : $matches[3];
322                 $deviceObj->device       = isset($deviceRegex['device_replacement']) ? str_replace("$1",$matches[1],$deviceRegex['device_replacement']) : str_replace("_"," ",$matches[1]);
323 
324                 // device version?
325                 $deviceObj->deviceVersion = isset($deviceObj->deviceMajor) ? $deviceObj->deviceMajor : "";
326                 $deviceObj->deviceVersion = isset($deviceObj->deviceMinor) ? $deviceObj->deviceVersion.'.'.$deviceObj->deviceMinor : $deviceObj->deviceVersion;
327 
328                 // prettify
329                 $deviceObj->deviceFull = $deviceObj->device." ".$deviceObj->deviceVersion;
330 
331                 // check to see if this is a mobile device
332                 // this isn't really needed because if it matches a mobile browser it'll automatically mark it as a mobile device
333                 $deviceObj->isMobileDevice = false;
334                 $mobileDevices  = array("iPhone","iPod","iPad","HTC","Kindle","Lumia","Amoi","Asus","Bird","Dell","DoCoMo","Huawei","i-mate","Kyocera",
335                     "Lenovo","LG","Kin","Motorola","Philips","Samsung","Softbank","Palm","HP ","Generic Feature Phone","Generic Smartphone",
336                     "Nintendo DSi","Nintendo 3DS","PlayStation Vita");
337                 foreach($mobileDevices as $mobileDevice) {
338                     if (stristr($deviceObj->device, $mobileDevice)) {
339                         $deviceObj->isMobileDevice = true;
340                         break;
341                     }
342                 }
343 
344                 // check to see if this is a tablet (not perfect)
345                 $deviceObj->isTablet = false;
346                 $tablets = array("Kindle","iPad","Playbook","TouchPad","Dell Streak","Galaxy Tab","Xoom");
347                 foreach($tablets as $tablet) {
348                     if (stristr($deviceObj->device, $tablet)) {
349                         $deviceObj->isTablet = true;
350                         break;
351                     }
352                 }
353 
354                 return $deviceObj;
355             }
356         }
357         return false;
358     }

375     /**
376      * Gets the latest user agent. Back-ups the old version first. it will fail silently if something is wrong...
377      */
378     public static function get() {
379         if ($data = @file_get_contents("https://raw.github.com/tobie/ua-parser/master/regexes.yaml")) {
380             if (file_exists(__DIR__."/resources/regexes.yaml")) {
381                 if (!self::$nobackup) { 
382                     if (!self::$silent) { print("backing up old YAML file...\n"); }
383                         if (!copy(__DIR__."/resources/regexes.yaml", __DIR__."/resources/regexes.".date("Ymdhis").".yaml")) {
384                             if (!self::$silent) { print("back-up failed...\n"); }
385                                 exit;
386                         }
387                 }
388             }
389             $fp = fopen(__DIR__."/resources/regexes.yaml", "w");
390             fwrite($fp, $data);
391             fclose($fp);
392             if (!self::$silent) { print("success...\n"); }
393         } else {
394             if (!self::$silent) { print("failed to get the file...\n"); }
395         }
396     }
