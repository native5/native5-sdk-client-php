 24 namespace Native5\Services\Account;
 25 
 26 /**
 27  * Application 
 28  */
 29 class Application
 30 {
 31     private $_name;
 32     private $_urls = array();
 33     private $_created;
 34     private $_modified;
 35     private $_repoConfig;
 36     private $_platforms;
 37     private $_deploymentConfigs = array();
 38 
 39     private $_releases;
 40 
 41     public function __construct($name) {
 42         $this->_name = $name;
 43     }
 44 
 45     public function setName($name) {
 46         $this->_name = $name;
 47     }
 48 
 49     public function getName() {
 50         return $this->_name;
 51     }
 52 
 53     public function setUrl($url) {
 54         if (!is_array($url))
 55             return false;
 56 
 57         $this->_urls = array_merge($this->_urls, $url);
 58         return true;
 59     }
 60 
 61     public function getUrl($env) {
 62         if (isset($this->_urls[$env]))
 63             return $this->_urls[$env];
 64         else
 65             return false;
 66     }
 67 
 68     public function setCreatedOn($dateTime) {
 69         $this->_created = $dateTime;
 70     }
 71 
 72     public function getCreatedOn() {
 73         return $this->_created;
 74     }
 75     
 76     public function setModifiedOn($dateTime) {
 77         $this->_modified = $dateTime;
 78     }
 79 
 80     public function getModifiedOn() {
 81         return $this->_modified;
 82     }
 83 
 84     public function setPlatforms($platforms) {
 85         $this->_platforms = $platforms;
 86     }
 87 
 88     public function getPlatforms() {
 89         return $this->_platforms;
 90     }
 91 
 92     public function setRepositoryConfig(ApplicationRepository $repoConfig) {
 93         $this->_repoConfig = $repoConfig;
 94     }
 95 
 96     public function getRepositoryConfig() {
 97         return $this->_repoConfig;
 98     }
 99 
100     public function addDeploymentConfig(ApplicationDeployment $deployConfig) {
101         $this->_deploymentConfigs[] = $deployConfig;
102     }
103 
104     public function getDeploymentConfigs() {
105         return $this->_deploymentConfigs;
106     }
107 
108     public function serialize($format) {
109         if($format == 'json') {
110             $output = array();
111             $output['name'] = $this->_name;
112             $output['created'] = $this->_created;
113             $output['modified'] = $this->_modified;
114             $output['platforms'] = $this->_platforms;
115             if (!empty($this->_repoConfig))
116                 $output['repo'] = $this->_repoConfig->serialize($format);
117             if (!empty($this->_deploymentConfigs)) {
118                 $output['deployConfig'] = array();
119                 foreach ($this->_deploymentConfigs as $deploy)
120                     $output['deployConfig'][] = $deploy->serialize($format);
121             }
122 
123             return $output;
124         }
125     }
126 }
