26 class ApplicationDAO {
27     public function buildApp($data) {
28         $app = new Application($data['name']);
29         if (isset($data['urls']) && !empty($data['urls'])) {
30             if (is_array($data['urls'])) {
31                 foreach ($data['urls'] as $env=>$url)
32                     $app->setUrl(array($env => $url));
33             } else {
34                 $app->setUrl(array('sandbox' => $data['urls']));
35             }
36         }
37         $created = (!empty($data['created']) ? $data['created'] : (!empty($data['created_on']) ? $data['created_on'] : null));
38         $modified = (!empty($data['modified']) ? $data['modified'] : (!empty($data['modified_on']) ? $data['modified_on'] : null));
39 
40         $app->setCreatedOn($created);
41         $app->setModifiedOn($modified);
42         $app->setPlatforms($data['platforms']);
43 
44         // Repository config
45         $appRepo = new ApplicationRepository();
46         $appRepo->setType($data['repository']['type']);
47         $appRepo->setUrl($data['repository']['url']);
48         $app->setRepositoryConfig($appRepo);
49 
50         // Deployment configurations - more than one for load balanced nodes
51         foreach ($data['deployment_metadata'] as $deploy) {
52             $appDeploy = new ApplicationDeployment();
53             $appDeploy->setServerName($deploy['name']);
54             $appDeploy->setUserName($deploy['user']);
55             $appDeploy->setDirectory($deploy['folder']);
56             $appDeploy->setEnvironment($deploy['environment']);
57             $app->addDeploymentConfig($appDeploy);
58         }
59 
60         return $app;
61     }
62 
63     private function _getStoreHandle() {
64         $client = new \MongoClient();
65         $db = $client->selectDB('native5');
66         return $db->selectCollection('accounts');
67     }
68 }
